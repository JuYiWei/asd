1.3 Date 对象

  • 属性
  • 方法
    • 普通时间
    • 世界时间
    • 格式化显示

属性

constructor 返回创建数组对象的原型函数。
prototype   允许你向数组对象添加属性或方法。

方法

普通时间

  • 获取
    var year = date.getFullYear();
    var month = date.getMonth();
    var dayOfMonth = date.getDate();
    var dayOfWeek = date.getDay();
    var hour = date.getHours();
    var minute = date.getMinutes();
    var second = date.getSeconds();
    var millisecond = date.getMilliseconds();
    var timestamp = date.getTime();
    var timestamp2 = date.valueOf();
    var zonneOffset = date.getTimezoneOffset();// 时差
    var timestampPadding = Date.parse("March 21, 2012");// 指定时间的时间戳

    console.log([year, month, dayOfMonth, dayOfWeek, hour, minute, second, millisecond, timestamp, zonneOffset]);
  • 设置
    date.setFullYear(2001);
    date.setMonth(4)
    date.setDate(5);
    date.setHours(12);
    date.setMinutes(13);
    date.setSeconds(14);
    date.setMilliseconds(132);
    
    date.setTime(13000000);

世界时间

  • 获取
    var yearUTC = date.getUTCFullYear();
    var monthUTC = date.getUTCMonth();
    var dayOfMonthUTC = date.getUTCDate();
    var dayOfWeekUTC = date.getUTCDay();
    var hourUTC = date.getUTCHours();
    var minuteUTC = date.getUTCMinutes();
    var secondUTC = date.getUTCSeconds();
    var millisecondUTC = date.getUTCMilliseconds();
    var timestampPaddingUTC = Date.UTC(2012,02,30,11,12,13,123);

    console.log([yearUTC, monthUTC, dayOfMonthUTC, dayOfWeekUTC, hourUTC, minuteUTC, secondUTC, millisecondUTC]);
  • 设置
    date.setUTCFullYear(2001);
    date.setUTCMonth(4)
    date.setUTCDate(5);
    date.setUTCHours(12);
    date.setUTCMinutes(13);
    date.setUTCSeconds(14);
    date.setUTCMilliseconds(132);

格式化显示

    var str1 = date.toDateString();
    var str2 = date.toISOString();
    var str3 = date.toJSON();
    var str4 = date.toLocaleDateString();
    var str5 = date.toLocaleTimeString();
    var str6 = date.toLocaleString();
    var str7 = date.toString();
    var str8 = date.toTimeString();
    var str9 = date.toUTCString();
